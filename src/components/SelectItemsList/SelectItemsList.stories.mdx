## Getting Started

```javascript
import React, { Component, useState } from 'react';
import { SelectItemsList } from '@codelittinc/agnostic-design-system';

const MySelectItemsList = () =>{
  const [selectedItem, setSelectedItem] = useState();

  const options = [
    { name: 'Software', id: 'Software' },
    { name: 'Financing', id: 'Financing' },
    { name: 'Retail', id: 'Retail' },
    { name: 'Pet', id: 'Pet' },
    { name: 'Law', id: 'Law' }
  ];

  render() {
    return (
      <SelectItemsList
        id={'select-items-list'}
        onChange={setSelectedItem}
        selected={selectedItem}
        options={options}
      />
    );
  }
}

export default MySelectItemsList;
```
